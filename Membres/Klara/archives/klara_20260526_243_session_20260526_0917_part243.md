def longueur_vers_coords(a, b, c, nom):
    # Construction d’un triangle à partir de ses trois côtés
    # a = BC, b = AC, c = AB (convention standard)
    # On place A en (0,0), B en (c,0), C à l’intersection de deux cercles
    xA, yA = 0, 0
    xB, yB = c, 0
    # distances : AC = b, BC = a
    # intersection des cercles (A,b) et (B,a)
    dx = xB - xA
    dy = yB - yA
    d = dx*dx + dy*dy
    if d < 1e-9:
        return None
    d = d**0.5
    if a + b < d or abs(a - b) > d:
        # triangle impossible (ne devrait pas arriver avec les données)
        return None
    # position de C
    xC = (b*b - a*a + c*c) / (2*c)
    yC = (b*b - xC*xC)**0.5
    return [(xA, yA), (xB, yB), (xC, yC)]
